New Age Category Column Function¶

def find_age_class(x):
    if x < 10:
        return "0-10"
    if x >= 10 and x < 20:
        return "10-20"
    if x >= 20 and x < 30:
        return "20-30"
    if x >= 30 and x < 40:
        return "30-40"
    if x >= 40 and x < 50:
        return "40-50"
    if x >= 50 and x < 60:
        return "50-60"
    if x >= 60:
        return "60+"
